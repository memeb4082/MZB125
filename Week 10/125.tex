    r &= \frac{(z+5)^3}{300}\\
    300r &= (z+5)^{3}\\
    \sqrt[3]{300r} &= z+5\\
    \sqrt[3]{300r} - 5 &= z\\
    z(r) &= \sqrt[3]{300r} - 5 r\in\bigg[r(0), r(1)\bigg]\\
    z(r) &= \sqrt[3]{300r} - 5 r\in\bigg[\frac{5}{12}, \frac{18}{25}\bigg]\\

    \subsection*{How much soil can the pot hold}
    V = $\int^{\infty}_{-\infty}A(z)dz \to \int^{\infty}_{-\infty}\pi r^{2}dz \to \pi \int^{\infty}_{-\infty} z(r)^{2} dr \therefore$
    \begin{flalign*}
        V &= \pi \int^{\frac{18}{12}}_{\frac{5}{12}} z(r)^{2} dr\\
        &= \pi \int^{\frac{18}{12}}_{\frac{5}{12}} \bigg(\sqrt[3]{300r}-5\bigg)^{2} dr\\
        &= \pi \int^{\frac{18}{12}}_{\frac{5}{12}} \bigg((300r)^{\frac{2}{3}}-10(300r)^{\frac{1}{3}}+25\bigg) dr\\
        &= \pi \bigg[300^{\frac{2}{3}}\times\frac{3}{5}r^{\frac{5}{3}}-10\times300^{\frac{1}{3}}\times\frac{3r^{\frac{4}{3}}}{4}+25r\bigg]^{\frac{18}{12}}_{\frac{5}{12}}\\
        &= \pi \bigg[\bigg(300^{\frac{2}{3}}\times\frac{3}{5}\times\frac{18}{12}^{\frac{5}{3}}-10\times300^{\frac{1}{3}}\times\frac{3\times\frac{18}{12}^{\frac{4}{3}}}{4}+25\times\frac{18}{12}\bigg) - \bigg(300^{\frac{2}{3}}\times\frac{3}{5}\times\frac{5}{12}^{\frac{5}{3}}-10\times300^{\frac{1}{3}}\times\frac{3\times\frac{5}{12}^{\frac{4}{3}}}{4}+25\times\frac{5}{12}\bigg)\bigg]\\
        &\simeq 9.74\text{m}^{3}
    \end{flalign*}
