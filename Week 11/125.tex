\section*{Question 9}
\begin{align*}
    z_{1} = 3+3i && z_{2} = 4-4i\\
\end{align*}
\begin{minipage}{0.48\linewidth}
    \begin{flalign*}
        \frac{1}{z_{1}} &= \frac{\tilde{z_{1}}}{|z_{1}|^{2}}\\
        &= \frac{3-3i}{(\sqrt{3^{2}+3^{2}})^{2}}\\
        &= \frac{3-3i}{3^{2}+3^{2}}\\
        &= \frac{3}{18} - \frac{3}{18}i\\
        &= \frac{1}{6} - \frac{1}{6}i
    \end{flalign*}
\end{minipage}
\begin{minipage}{0.48\linewidth}
    \begin{flalign*}
        \frac{1}{z_{2}} &= \frac{\tilde{z_{2}}}{|z_{2}|^{2}}\\
        &= \frac{4+4i}{(\sqrt{4^{2} + (-4)^{2}})^{2}}\\
        &= \frac{4+4i}{16+16}\\
        &= \frac{4}{32}+\frac{4}{32}i\\
        &= \frac{1}{8}+\frac{1}{8}i
    \end{flalign*}
\end{minipage}
\begin{flalign*}
    \frac{1}{z} &= \frac{1}{z_{1}}+\frac{1}{z_{2}}\\
    &= \bigg(\frac{1}{6} - \frac{1}{6}i\bigg) + \bigg(\frac{1}{8}+\frac{1}{8}i\bigg)\\
    &= \bigg(\frac{1}{6}+\frac{1}{8}\bigg) + \bigg(\frac{1}{8}-\frac{1}{6}\bigg)i\\
    &= \bigg(\frac{8}{48}+\frac{6}{48}\bigg) + \bigg(\frac{6}{48}-\frac{8}{48}\bigg)i\\
    &= \frac{14}{48} - \frac{2}{48}i\\
    &= \frac{7}{24} - \frac{1}{24}i
\end{flalign*}
\begin{flalign*}
    \frac{1}{\frac{1}{z}} = \frac{\tilde{\frac{1}{z}}}{|\frac{1}{z}|^{2}} = z
\end{flalign*}
\begin{flalign*}
    \therefore z &= \frac{\frac{7}{24} + \frac{1}{24}i}{(\sqrt{\frac{7}{24}^{2} + \frac{1}{24}^{2}})^{2}}\\
    &= \frac{\frac{7}{24} + \frac{1}{24}i}{\frac{49}{576} + \frac{1}{576}}\\
    &= \frac{\frac{7}{24} + \frac{1}{24}i}{\frac{50}{576}}\\
    &= \frac{288}{25}\bigg(\frac{7}{24}+\frac{1}{24}i\bigg)\\
    &= \frac{12}{25}\bigg(7+i\bigg)\\
    &= \frac{84}{25}+\frac{12}{25}i\\
    \therefore |z| &= \sqrt{\frac{84}{25}^{2}+\frac{12}{25}^{2}} = \frac{12\sqrt{2}}{5}\\
    \tan(\theta) &= \frac{\frac{12}{25}}{\frac{84}{25}}\\
    &= \frac{1}{7}\\
    \theta &= \arctan\bigg(\frac{1}{7}\bigg)\\
    &\approx 0.142
\end{flalign*}
